1 /*
2  * Class: MecanimSDX
3  * Author: sphereii, Mortelentus
4  * Category: Mecanim Animator
5  * Description:
6  *
7  * This class provides mecanim animations to be applied to custom entities. It allows us to use Unity Animator State Machine to make
8  * more complex animation sequences. RootMotion is NOT supported, and must be disabled.
9  *
10  * Mecanim allows us to create complex state machines in unity, to allow more variety of animations. For example, we could have 8 different animations for
11  * an attack animation. This class, when properly configured using the AttackIndexes of 8, will randomly pick one of the attacks to display.
12  *
13  * Special thanks to Mortelentus, who added in the additional code to allow custom entities to use ranged weapons.
14  *
15  * Usage:
16  * Add the following class to entities that are meant to use these features.
17  *
18  * <property name="AvatarController" value="MecanimSDX, Mods" />
19  *
20  *
21  * Features:
22  * This class is meant to be used in following the Mecanim Tutorial:https://7d2dsdx.github.io/Tutorials/HowtocreateanAnimatorStateMachin.html
23  *
24  * Because Mecanim is so flexible, and we cannot attach scripts to the unity objects, this class has quite a few available XML values to help tweak each entity
25  * without adjusting this class.
26  *
27  * This tag is mandatory and must contain the animator's state names for each possible attack.
28  * <property name="AttackAnimations" value="Attack0, Attack1" />
29  *
30  * The following indexes default to 0, so they do not need to be included if you only have one animation for a particular state. For example, if you only have one Attack,
31  * you would not need to set AttackIndexes value at all. If you have two attacks, you would have to set AttackIndexes to 2.
32  *
33  * The following example has two different attacks, 2 different run and walk animations each:
34  * <property name="AttackIndexes" value="2" />
35  * <property name="SpecialAttackIndexes" value="0" />
36  * <property name="PainIndexes" value="0" />
37  * <property name="DeathIndexes" value="0" />
38  * <property name="RunIndexes" value="2" />
39  * <property name="WalkIndexes" value="2" />
40  * <property name="IdleIndexes" value="0" />
41  * <property name="JumpIndexes" value="0" />
42  *
43  * Not all indexes are currently used or called yet, but future revisions will support them, such as Stun, Crouch, Electrocution, Raging, SpecialSecondAttack, harvest, and eating.
44  *
45  * <property name="SpecialSecondAttackIndexes" value="0" />
46  * <property name="RagingIndexes" value="0" />
47  * <property name="ElectrocutionIndexes" value="0" />
48  * <property name="CrouchIndexes" value="0" />
49  * <property name="StunIndexes" value="0" />
50  * <property name="SleeperIndexes" value="0" />
51  * <property name="HarvestIndexes" value="0" />
52  *
53  * A special RandomIndexes is supplied, which can be be re-used in the State Machine to add further randomized of attacks.
54  */

592  Log("Animation Type found: " + entityClass.Properties.Values[strAnimationType].ToString());

595  Log("Adding Attack Hash: " + strAnimationState.Trim());
